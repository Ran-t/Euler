Mults :: (Int a) => [a] -> [a]

Mults [] = []

Mults (x:xs)
        | mod x 3 == 0        = x : Mults xs
        | mod x 5 == 0        = x : Mults xs
        | otherwise           = Mults xs

sum (Mults [1..1000])
